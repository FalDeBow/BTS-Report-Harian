    function kirimKeGoogleSheet() {

      const tanggal = document.getElementById('tanggal').value;

      const namaPetugas = document.getElementById('namaPetugas').value;

      const jamStart = document.getElementById('jamStart').value;

      const jamFinish = document.getElementById('jamFinish').value;


      if (!namaPetugas || !jamStart) {

        alert("Harap isi Nama Petugas dan Jam Start!");

        return;

      }


      const titikBoxes = document.querySelectorAll('.titik-box');

      let arrayTitik = [];

      let teksDaftarTitik = "";


      titikBoxes.forEach((box, index) => {

        const namaSite = box.querySelector('.site-nama').value;

        const jamTiba = box.querySelector('.site-tiba').value;

        const jamSelesai = box.querySelector('.site-selesai').value;

        const aktivitas = box.querySelector('.site-aktivitas').value;

        const status = box.querySelector('.site-status').value;


        if (namaSite) {

          let durasi = "-";

          if (jamTiba && jamSelesai) {

            const [h1, m1] = jamTiba.split(':');

            const [h2, m2] = jamSelesai.split(':');

            const diffMin = (parseInt(h2)*60 + parseInt(m2)) - (parseInt(h1)*60 + parseInt(m1));

            durasi = (diffMin / 60).toFixed(2);

          }


          arrayTitik.push({

            namaSite: namaSite,

            jamTiba: jamTiba,

            jamSelesai: jamSelesai,

            durasi: durasi,

            aktivitas: aktivitas,

            status: status

          });


          teksDaftarTitik += `${index + 1}. ${namaSite} (${jamTiba}-${jamSelesai}) - Status: ${status} | ${aktivitas}\n`;

        }

      });


      const teksLaporan = `*LAPORAN PERJALANAN HARIAN BTS*\n` +

                          `-----------------------------------\n` +

                          `📅 Tanggal  : ${tanggal}\n` +

                          `👤 Petugas  : ${namaPetugas}\n` +

                          `🕒 Jam Start: ${jamStart} WIB\n` +

                          `🏁 Jam Finish: ${jamFinish} WIB\n` +

                          `-----------------------------------\n` +

                          `📍 *DAFTAR TITIK LOKASI:*\n` +

                          `${teksDaftarTitik || '- Tidak ada titik\n'}` +

                          `-----------------------------------`;


      document.getElementById('hasilTeks').value = teksLaporan;


      const payload = {

        tanggal: tanggal,

        namaPetugas: namaPetugas,

        jamStart: jamStart,

        jamFinish: jamFinish,

        titikLokasi: arrayTitik

      };


      alert("Mengirim data ke database Admin...");


      fetch(SCRIPT_URL, {

        method: "POST",

        mode: "no-cors",

        headers: { "Content-Type": "application/json" },

        body: JSON.stringify(payload)

      })

      .then(() => {

        alert("✅ Berhasil dikirim ke Google Sheet Admin!");

      })

      .catch(error => {

        alert("⚠️ Gagal mengirim data: " + error);

      });

    }


    function salinTeks() {

      const teks = document.getElementById('hasilTeks');

      teks.select();

      navigator.clipboard.writeText(teks.value);

      alert('Teks laporan berhasil disalin!');

    }
